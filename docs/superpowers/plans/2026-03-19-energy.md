# Energy System Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add blue/red energy tracking per character, tier-based energy task execution, energy display on character cards and login cards, and pre-seeded default energy tasks.

**Architecture:** Backend handlers manage energy state server-side (with authoritative timestamps). Renderer computes current energy from stored current + elapsed regen. Default tasks are seeded on startup via slug-based deduplication. No separate branches — single commit when all changes are complete and functional.

**Tech Stack:** Electron 41, electron-store, vanilla JS renderer, IPC via contextBridge.

**Spec:** `docs/superpowers/specs/2026-03-19-energy-design.md`

- [ ] **Create `src/main/handlers/energy.js`**

```js
const { ipcMain } = require('electron');
const store = require('../store');

function serverComputeEnergy(stored) {
  if (!stored) return 0;
  const { current, max, regenMin, lastUpdated } = stored;
  const gained = Math.floor((Date.now() - lastUpdated) / (regenMin * 60000));
  return Math.min(max, Math.max(0, current + gained));
}

module.exports = function registerEnergyHandlers() {
  ipcMain.handle('energy:runTask', (_, characterId, taskId, tierIndex) => {
    const task = store.get('tasks').find(t => t.id === taskId);
    if (!task || !task.tiers || !task.tiers[tierIndex]) return store.get('characters');
    const cost = task.tiers[tierIndex].energyCost;
    const updated = store.get('characters').map(c => {
      if (c.id !== characterId) return c;
      const available = serverComputeEnergy(c.blueEnergy);
      if (available < cost) return c;
      return {
        ...c,
        blueEnergy: { ...c.blueEnergy, current: Math.max(0, available - cost), lastUpdated: Date.now() },
        runCounts: { ...(c.runCounts || {}), [String(taskId)]: ((c.runCounts || {})[String(taskId)] || 0) + 1 },
      };
    });
    store.set('characters', updated);
    return updated;
  });

  ipcMain.handle('energy:runRedTask', (_, characterId, taskId, tierIndex) => {
    const task = store.get('tasks').find(t => t.id === taskId);
    if (!task || !task.tiers || !task.tiers[tierIndex]) return store.get('characters');
    const cost = task.tiers[tierIndex].energyCost;
    const updated = store.get('characters').map(c => {
      if (c.id !== characterId) return c;
      const available = serverComputeEnergy(c.redEnergy);
      if (available < cost) return c;
      return {
        ...c,
        redEnergy: { ...c.redEnergy, current: Math.max(0, available - cost), lastUpdated: Date.now() },
        taskState: { ...c.taskState, [String(taskId)]: true },
      };
    });
    store.set('characters', updated);
    return updated;
  });
};
```

  ```js
  // ─── Energy helpers ───────────────────────────────────────────────────────────
  // Computes current energy from stored snapshot + elapsed regen time.
  // Returns null if energy not configured. Clamps to [0, max].
  function computeEnergy(stored, now = Date.now()) {
    if (!stored) return null;
    const { current, max, regenMin, lastUpdated } = stored;
    const gained = Math.floor((now - lastUpdated) / (regenMin * 60000));
    return Math.min(max, Math.max(0, current + gained));
  }
  ```
